1. [Write a program to create an ArrayList, ask the user for `n`, take `n` integers from the user, store them in the list, and print the ArrayList.](#problem1)
2. [Write a program to create an ArrayList, ask the user for `n` and read `n` integers, then ask the user for an index and a value (both taken from the user) and insert that value at the given index; finally print the updated list.](#problem2)
3. [Write a program to create an ArrayList, ask the user for `n` and read `n` integers, then ask the user for a value to remove (taken from the user) and remove it by value, then ask the user for an index (taken from the user) and remove the element at that index; print the list after each removal.](#problem3)
4. [Write a program to create an ArrayList, ask the user for `n` and read `n` integers, then ask the user for an index and a new value (both from the user) and replace the element at that index with the new value; print the result.](#problem4)

## Solutions:
#### problem1
```java
package arrayListSolutions;

import java.util.Scanner;
import java.util.ArrayList;

public class AddElements {

	public static void main(String[] args) {
		Scanner input = new Scanner(System.in);
		
		System.out.print("number of input: ");
		int size = input.nextInt();
		
		ArrayList<Integer> al = new ArrayList<Integer>();
		
		System.out.print("Enter " + size + " integers separated by a whitespace: ");
		for(int i=0; i<size; i++ ) {
			al.add(input.nextInt());
		}
		
		System.out.println("Array Lists are: " + al);
		input.close();
	}

}
```

#### problem3
```java
package arrayListSolutions;

import java.util.Scanner;
import java.util.ArrayList;

public class RemoveElementAtIndex {

	public static void main(String[] args) {
		Scanner input = new Scanner(System.in);
		
		System.out.print("number of input: ");
		int size = input.nextInt();
		
		ArrayList<Integer> al = new ArrayList<Integer>();
		
		System.out.print("Enter " + size + " integers separated by a whitespace: ");
		for(int i=0; i<size; i++)  {
			al.add(input.nextInt());
		}
		
		System.out.println("Array List: " + al);
		
		System.out.print("Give the value to be remove: ");
		int value = input.nextInt();
		
		// remove all the occurrences of value from arrayList
		for(int i=0; i<al.size(); i++) {
			if(al.get(i) == value) {
				al.remove(i);
				i--;
			}
		}
		
		// print arrayList after removing
		System.out.println("After removal of all occurences of '" + value + "', Updated Array List: " + al);
		
		// remove the value from 
		System.out.print("Enter the index no. where you want to delete a value: ");
		int index = input.nextInt();
		
		al.remove(index);
		System.out.print("After deletion at index '" + index + "', new updated Array List: " + al);
		
		input.close();
	}

}
```

#### problem4
```java
package arrayListSolutions;

import java.util.Scanner;
import java.util.ArrayList;

public class ReplaceElementAtIndex {

	public static void main(String[] args) {
		Scanner input = new Scanner(System.in);
		
		System.out.print("number of input: ");
		int size = input.nextInt();
		
		ArrayList<Integer> al = new ArrayList<Integer>();
		
		// user-input to fill the ArrayList
		System.out.print("Enter " + size + " integers separated by a whitespace: ");
		for(int i=0; i<size; i++) {
			al.add(input.nextInt());
		}
		
		// print the ArrayList
		System.out.println("Array List: " + al);
		
		// insert the index
		System.out.print("Enter the index for replacement: ");
		int index = input.nextInt();
		
		// check number of index is valid 
		if(index < 0 || index >= al.size()) {
			System.out.print("Invalid index, No replacement done.");
		} else {
			System.out.print("Enter the value after replacement: ");
			int value = input.nextInt();
			al.set(index, value);
			System.out.println("After replacement, updated ArrayList: " + al);
		}
		
		
		
		
		input.close();
	}

}
```
